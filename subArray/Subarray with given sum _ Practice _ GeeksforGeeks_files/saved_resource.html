class Solution

{

    //Function to find a continuous sub-array which adds up to a given number.

    static ArrayList<Integer> subarraySum(int[] arr, int n, int s) 

    {

        ArrayList<Integer> ret=new ArrayList<Integer>();

        int i=0,j=0,sum=arr[0];

        if(s==0){

            ret.add(-1);

            return ret;

        }

        else{

        while(j<n){

        if(sum==s){

            ret.add(i+1);

            ret.add(j+1);

            return ret;

        }

        else{

        if(sum<s){

            j++;

            if(j>=n){

                break;

            }

            else{

            sum=sum+arr[j];

            continue;

            }

        }

        if(sum>s){

            sum=sum-arr[i];

            i++; 

            continue;

        }

        }

        }

        ret.add(-1);

        return ret;

        }

    }

    

}

    static ArrayList<Integer> subarraySum(int[] arr, int n, int s) 
    {
        // Your code here
        ArrayList<Integer> list=new ArrayList<>();
        long sum=0;
        int j=-1;
        int i;
        int p=1;
        if(s==0){
          list.add(-1);
          return list;
        }
        for(i=0;i<n;i++)
        {
            sum+=arr[i];
            if(sum>s)
            {
                while(sum>s)
                {
                    j++;
                    sum-=arr[j];
                }
            }
            if(sum==s)
            {
                p=0;
             break;
            }
            
        }
        
        if(p==0){
        list.add(j+2);
        list.add(i+1);
        return list;
        }

        {
            list.add(-1);
            return list;
        }
    }

This problem can be solved by using Sliding window.

(we can use sliding window since it is told in question that it contains non- negative integers)

note:- if the question would have mentioned  that it contains -ve integers then we would not able to use sliding window.

code--

vector<int> subarraySum(vector<int>arr, int n, long long s)

    {

      int i=0,j=0;

      long long sum=0;

      

      vector<int>v;

      

      while(j<n){

          sum+=arr[j];

          while(sum>s){

              sum-=arr[i];

              i++;

          }

      

       if(sum==s && i<=j){

              v.push_back(i+1);

              v.push_back(j+1);

              return v;

             }

       j++;

    }   

     /*note -one can tell why i<=j condition is used . It is used because 

in the problem it is stated that --In case of multiple subarrays, return the subarray indexes which comes first on moving from left to right.

*/

     v.push_back(-1);

     return v;

  }

vector<int> subarraySum(vector<int>arr, int n, long long s)
    {
        // Your code here
        int first_index  = 0;
        int second_index = 0;
        
        long long sum = 0;
        
        if(s == 0) 
            return {-1};
        
        while(second_index <= n && first_index <= second_index) {
            
            if(sum > s) {
                sum -= arr[first_index];
                first_index++;
            }
            
            else if(sum < s) {
                sum += arr[second_index];
                ++second_index;
            }
            
            else {
                return {first_index+1, second_index};
            }
        }
        
        return {-1};
